    <script>
        // --- 1. Three.js Background Hearts ---
        let scene, camera, renderer, hearts = [];
        const heartCount = 25;

        function initThree() {
            scene = new THREE.Scene();
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('three-canvas'), alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);

            const heartShape = new THREE.Shape();
            const x = 0, y = 0;
            heartShape.moveTo(x + 5, y + 5);
            heartShape.bezierCurveTo(x + 5, y + 5, x + 4, y, x, y);
            heartShape.bezierCurveTo(x - 6, y, x - 6, y + 7, x - 6, y + 7);
            heartShape.bezierCurveTo(x - 6, y + 11, x - 3, y + 15.4, x + 5, y + 19);
            heartShape.bezierCurveTo(x + 12, y + 15.4, x + 16, y + 11, x + 16, y + 7);
            heartShape.bezierCurveTo(x + 16, y + 7, x + 16, y, x + 10, y);
            heartShape.bezierCurveTo(x + 7, y, x + 5, y + 5, x + 5, y + 5);

            const geometry = new THREE.ExtrudeGeometry(heartShape, { depth: 2, bevelEnabled: true, bevelThickness: 1, bevelSize: 1 });
            const material = new THREE.MeshPhongMaterial({ color: 0xff3366, shininess: 100 });
            
            const light = new THREE.DirectionalLight(0xffffff, 1);
            light.position.set(1, 1, 1).normalize();
            scene.add(light);
            scene.add(new THREE.AmbientLight(0x404040));

            for (let i = 0; i < heartCount; i++) {
                const heart = new THREE.Mesh(geometry, material);
                resetHeart(heart);
                scene.add(heart);
                hearts.push(heart);
            }

            camera.position.z = 100;
        }

        function resetHeart(heart) {
            heart.position.set((Math.random() - 0.5) * 200, (Math.random() - 0.5) * 200, (Math.random() - 0.5) * 100);
            heart.scale.set(0.5, 0.5, 0.5);
            heart.rotation.set(Math.random() * Math.PI, Math.random() * Math.PI, 0);
            heart.userData = { 
                speed: 0.05 + Math.random() * 0.1,
                wave: Math.random() * 5,
                waveSpeed: 0.02 + Math.random() * 0.03
            };
        }

        function animateThree() {
            requestAnimationFrame(animateThree);
            hearts.forEach(heart => {
                heart.position.y += heart.userData.speed;
                heart.position.x += Math.sin(Date.now() * heart.userData.waveSpeed) * 0.1;
                heart.rotation.y += 0.01;
                if (heart.position.y > 100) heart.position.y = -100;
            });
            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        // --- 2. Step Navigation ---
        let currentStep = 1;

        function nextStep(step) {
            const currentEl = document.getElementById(`step${currentStep}`);
            const nextEl = document.getElementById(`step${step}`);
            
            gsap.to(currentEl.querySelector('.step-content'), { opacity: 0, y: -20, duration: 0.5 });
            gsap.to(currentEl, { scale: 0.9, opacity: 0, duration: 0.5, onComplete: () => {
                currentEl.style.display = 'none';
                nextEl.style.display = 'block';
                
                // Update Progress Bar
                document.getElementById('progress-fill').style.width = `${(step/5)*100}%`;
                
                gsap.fromTo(nextEl, { scale: 1.1, opacity: 0 }, { scale: 1, opacity: 1, duration: 0.6 });
                gsap.fromTo(nextEl.querySelector('.step-content'), { opacity: 0, y: 20 }, { opacity: 1, y: 0, duration: 0.6, delay: 0.2 });
            }});
            currentStep = step;
        }

        // --- 3. Polaroid Parallax ---
        const polaroid = document.getElementById('memory-polaroid');
        document.addEventListener('mousemove', (e) => {
            if (currentStep !== 4) return;
            const x = (window.innerWidth / 2 - e.pageX) / 25;
            const y = (window.innerHeight / 2 - e.pageY) / 25;
            polaroid.style.transform = `rotateY(${x}deg) rotateX(${y}deg) rotateZ(-3deg)`;
        });

        // --- 4. Grand Finale ---
        function startFinale() {
            const btn = document.getElementById('celebrate-btn');
            const finalMsg = document.getElementById('final-message');
            
            btn.style.pointerEvents = 'none';
            gsap.to(btn, { opacity: 0, scale: 0, duration: 0.5, onComplete: () => btn.classList.add('hidden') });
            
            setTimeout(() => {
                finalMsg.classList.remove('hidden');
                gsap.fromTo(finalMsg, { opacity: 0, y: 30 }, { opacity: 1, y: 0, duration: 1.5, ease: "power4.out" });
                
                // Confetti Explosion
                const duration = 5 * 1000;
                const animationEnd = Date.now() + duration;
                const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 0 };

                const interval = setInterval(function() {
                    const timeLeft = animationEnd - Date.now();
                    if (timeLeft <= 0) return clearInterval(interval);
                    
                    const particleCount = 50 * (timeLeft / duration);
                    confetti({ ...defaults, particleCount, origin: { x: Math.random(), y: Math.random() - 0.2 } });
                    confetti({ 
                        ...defaults, 
                        particleCount: 10, 
                        shapes: ['circle'], 
                        colors: ['#FFC0CB', '#FF69B4', '#FF1493'],
                        scalar: 1.2
                    });
                }, 250);

                // Final Heart Animation
                hearts.forEach(heart => {
                    gsap.to(heart.position, {
                        y: 300,
                        x: (Math.random() - 0.5) * 500,
                        z: 200,
                        duration: 3,
                        ease: "power2.in"
                    });
                    gsap.to(heart.material, { opacity: 0, duration: 3 });
                });
            }, 500);
        }

        // Init
        initThree();
        animateThree();
        gsap.to('.step-content', { opacity: 1, y: 0, duration: 1, delay: 0.5 });
    </script>
